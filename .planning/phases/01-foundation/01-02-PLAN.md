Create `src/Types.luau` with `--!strict` at the top. Define complete type interfaces for all three builders plus callback types. These are the contracts Phase 2 must implement.

Per user decision: full interface definitions in Phase 1, strict callback types, individual type exports, direct Roblox engine types in signatures.

```lua
--!strict

-- Callback types with strict signatures
export type OnPressed = (actionName: string, input: InputObject) -> ()
export type OnReleased = (actionName: string, input: InputObject) -> ()
export type OnStateChanged = (actionName: string, newState: boolean, input: InputObject) -> ()

-- Forward declarations needed for cross-references
export type ContextBuilder = {
    action: (self: ContextBuilder, name: string) -> ActionBuilder,
    enable: (self: ContextBuilder) -> ContextBuilder,
    disable: (self: ContextBuilder) -> ContextBuilder,
    setPriority: (self: ContextBuilder, priority: number) -> ContextBuilder,
    setSink: (self: ContextBuilder, sink: boolean) -> ContextBuilder,
    destroy: (self: ContextBuilder) -> (),
    getRawInstance: (self: ContextBuilder) -> InputContext,
}

export type ActionBuilder = {
    bind: (self: ActionBuilder, keyCode: Enum.KeyCode) -> BindingBuilder,
    onPressed: (self: ActionBuilder, callback: OnPressed) -> ActionBuilder,
    onReleased: (self: ActionBuilder, callback: OnReleased) -> ActionBuilder,
    onStateChanged: (self: ActionBuilder, callback: OnStateChanged) -> ActionBuilder,
    context: (self: ActionBuilder) -> ContextBuilder,
    getRawInstance: (self: ActionBuilder) -> InputAction,
}

export type BindingBuilder = {
    withModifier: (self: BindingBuilder, modifier: Enum.KeyCode) -> BindingBuilder,
    withSecondaryModifier: (self: BindingBuilder, modifier: Enum.KeyCode) -> BindingBuilder,
    action: (self: BindingBuilder) -> ActionBuilder,
    getRawInstance: (self: BindingBuilder) -> InputBinding,
}

return {}
```

Key points:
- ActionBuilder has a `context()` method to navigate back up the chain (enables fluent `.bind().action().onPressed()` patterns)
- BindingBuilder has an `action()` method for the same back-reference reason
- All methods that return `self` use the concrete type (ContextBuilder, ActionBuilder, BindingBuilder) for proper chaining autocomplete
- getRawInstance() on each builder per user decision: "underlying IAS Instances hidden by default, with :getRawInstance() escape hatch"
- No `any` types anywhere — every parameter and return is strictly typed

**src/IASAdapter.luau** — Thin adapter wrapping all IAS Instance.new() calls. Per user decision: "a single module wrapping Instance.new() calls; if IAS API changes, one file gets updated." Adapter creates real IAS Instances (functional from day one), not placeholders.

```lua
--!strict
local IASAdapter = {}

function IASAdapter.newContext(name: string): InputContext
    local context = Instance.new("InputContext")
    context.Name = name
    context.Enabled = false  -- disabled until explicitly activated
    return context
end

function IASAdapter.newAction(name: string, actionType: Enum.InputActionType): InputAction
    local action = Instance.new("InputAction")
    action.Name = name
    action.Type = actionType
    return action
end

function IASAdapter.newBinding(keyCode: Enum.KeyCode): InputBinding
    local binding = Instance.new("InputBinding")
    binding.KeyCode = keyCode
    return binding
end

return IASAdapter
```
